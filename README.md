# Identificador-de-n-meros-pares-impares

par = []
impar = []

for i in range(2):
    numero = int(input('Digite um numero: ') 
    if numero % 2 == 0:
    par.append(numero)

    else:
        impar.append(numero)

print('\nOs numeros pares são:')
for numero in par:
    print(f' - {numero}')

print('\nOs numeros impares são:')
for numero in impar:
    print(f' - {numero}')
